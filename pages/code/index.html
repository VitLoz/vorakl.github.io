#!./dumb-init /bin/bash

# rc - the basic Entrypoint
# Written by oleksii.tsvietnov@zooplus.com

# --[ Environment ]--
# RC_DEBUG (true|false) [false]
# RC_BG_WAIT_POLICY (wait_bg_all|wait_bg_any|wait_bg_none) [wait_bg_none]
# RC_FG_WAIT_POLICY (wait_fg_none|wait_fg_yes) [wait_fg_none]

set +e # Do not exit on errors by default
[ "$RC_DEBUG" = "true" ] && { set -x; } # Turn on Debug mode if true
: ${RC_BG_WAIT_POLICY:=wait_bg_none} # Set default value for a wait bg policy
: ${RC_FG_WAIT_POLICY:=wait_fg_none} # Set default value for a wait fg policy


main() {
    trap hook_main_exit EXIT
    trap 'exit' INT

    SELF_NAME=$(basename -s .sh $0) # Self name for logging purpose
    CHILDPID="" # Collecting child processes
    NS="main" # Name Space
    local wait_bg_cmd

    case "$RC_BG_WAIT_POLICY" in
        wait_bg_all)  wait_bg_cmd="wait";;
        wait_bg_any)  wait_bg_cmd="wait -n";;
        wait_bg_none) wait_bg_cmd="";;
        *)            wait_bg_cmd="$RC_BG_WAIT_POLICY";;
    esac

    say "The default waiting policy for bg processes: $RC_BG_WAIT_POLICY"
    say "The default waiting policy for a fg process: $RC_FG_WAIT_POLICY"

    # Check for Background tasks
    while [ "$1" = "-D" ]; do
        shift
        (
            trap 'hook_sub_exit "$1"' EXIT

            NS="bg"

            set -e # Exit on errors in the sub-shell
            say "Running on the background: $1"
            eval "$1"
        )&
        CHILDPID="${CHILDPID} $!"
        shift
    done

    # Check for a foreground task
    if [ -n "$*" ]; then
        (
            trap 'hook_sub_exit "$@"' EXIT

            NS="fg"

            set -e # Exit on errors in the sub-shell
            say "Running on the foreground: $@"
            eval "$@"
        )
    fi

    $wait_bg_cmd
}

say() {
    echo "$(date '+%Y-%m-%d %H:%M:%S') ($SECONDS sec) ${SELF_NAME} [$NS]: $@"
}

warn() {
    say "$@" >&2
}

hook_main_exit() {
    local pid

    say "Handling of termination..."
    for pid in $CHILDPID; do
        ps -p $pid &> /dev/null && { say " - terminating $pid"; kill -15 $pid; } || true
    done
    say "Exited."
}

hook_sub_exit() {
    local rc=$?
    local ns

    case "$NS" in
        fg) ns="foreground";;
        bg) ns="background";;
        *)  ns="$NS";;
    esac

    say "Exiting on the $ns ($rc): $@"
#    [ "$NS" = "bg" -a "$RC_FG_WAIT_POLICY" = "wait_fg_none" ] && exit 128 || return $rc
    return $rc
}

main "$@"

#!/usr/bin/env python3

import requests, sys

environment = sys.argv[1] if len(sys.argv) > 1 else 'dev'

marathon_request_url = 'http://marathon.{}.zooplus.net:8080/v2/tasks'.format(environment)


def apps_grouped_by_hosts():

    apps_by_hosts = dict()
    try:
        tasks = requests.get(marathon_request_url).json()
    except requests.exceptions.RequestException as err:
        print(err, file=sys.stderr)
        raise SystemExit(0)

    for task in tasks['tasks']:
        if task['state'] == 'TASK_RUNNING':
            if task['host'] not in apps_by_hosts: 
                apps_by_hosts[ task['host'] ] = list()
            apps_by_hosts[ task['host'] ].append( (task['id'], task['appId']) )

    return apps_by_hosts

if __name__ == '__main__':
    apps = apps_grouped_by_hosts()

    for host in apps:
        print()
        print(host)
        for app in apps[host]:
            print('  {}\t{}'.format(app[1],app[0]))
